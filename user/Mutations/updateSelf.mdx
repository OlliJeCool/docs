---
title: 'updateSelf'
description: 'Updates the information of the currently authenticated user.'
---
    
    ## Arguments
    
    <ParamField body="newUserData" type="UpdateUserInput">
    Input object for updating user information.
    <Expandable title="fields">
    <ResponseField name="about" type="String">New about the user text to update.</ResponseField>
<ResponseField name="from" type="String">A new user location.</ResponseField>
<ResponseField name="name" type="String">A new name.</ResponseField>
    </Expandable>
    </ParamField> 

    
    ## Returns

    <ParamField body="updateSelf" type="User">
    User model
    <Expandable title="fields">
    <ResponseField name="about" type="String">Information about the user.</ResponseField>
<ResponseField name="backgrounds" type="[Background!]!">Backgrounds owned by user.</ResponseField>
<ResponseField name="followedBy" type="[User!]!">Users following user.</ResponseField>
<ResponseField name="followers" type="FollowingCounts!"/>
<ResponseField name="following" type="[User!]!">Users followed by user.</ResponseField>
<ResponseField name="from" type="String">User location.</ResponseField>
<ResponseField name="id" type="String!">UUID generated by Auth0 and assigned on creation.</ResponseField>
<ResponseField name="name" type="String!">User name, not unique.</ResponseField>
<ResponseField name="products" type="[Product!]!">Products owned by user.</ResponseField>
<ResponseField name="selectedBackground" type="Background!">The background selected by the user to be displayed on their card.</ResponseField>
<ResponseField name="socials" type="[Social!]!">Socials owned by user.</ResponseField>
<ResponseField name="verified" type="Boolean!">Defines if the user is verified by Decode Fashion.</ResponseField>
    </Expandable>
    </ParamField>
